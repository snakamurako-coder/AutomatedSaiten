/** OpenCV 透視補正・OMR 共通ロジック */ var CELL_PX = 20; var OFFSET_COL = 2; var OFFSET_ROW = 2; function getPaperConfig(orientation) { var isPortrait = orientation === 'portrait'; return { isPortrait: isPortrait, borderCols: isPortrait ? 51 : 73, borderRows: isPortrait ? 73 : 51, warpW: (isPortrait ? 51 : 73) * CELL_PX, warpH: (isPortrait ? 73 : 51) * CELL_PX, idStartCol: isPortrait ? 42 : 64, idStartRow: 3 }; } function warpImageFromElement(imgElement, orientation, threshVal) { var cfg = getPaperConfig(orientation); threshVal = threshVal != null ? threshVal : 128; var src = cv.imread(imgElement); var gray = new cv.Mat(); var thresh = new cv.Mat(); var contours = new cv.MatVector(); var hierarchy = new cv.Mat(); cv.cvtColor(src, gray, cv.COLOR_RGBA2GRAY, 0); cv.threshold(gray, thresh, threshVal, 255, cv.THRESH_BINARY_INV); cv.findContours(thresh, contours, hierarchy, cv.RETR_EXTERNAL, cv.CHAIN_APPROX_SIMPLE); var maxArea = 0, maxContour = null; for (var i = 0; i < contours.size(); ++i) { var cnt = contours.get(i); var area = cv.contourArea(cnt); if (area > maxArea) { maxArea = area; maxContour = cnt; } } if (!maxContour) { src.delete(); gray.delete(); thresh.delete(); contours.delete(); hierarchy.delete(); throw new Error('用紙外周の黒い太枠線を検知できません。'); } var approx = new cv.Mat(); cv.approxPolyDP(maxContour, approx, 0.02 * cv.arcLength(maxContour, true), true); if (approx.rows !== 4) { src.delete(); gray.delete(); thresh.delete(); contours.delete(); hierarchy.delete(); approx.delete(); throw new Error('四角形外枠の検出に失敗しました。'); } var pts = []; for (var j = 0; j < 4; j++) pts.push({ x: approx.data32S[j * 2], y: approx.data32S[j * 2 + 1] }); pts.sort(function(a, b) { return a.y - b.y; }); var topPts = pts.slice(0, 2).sort(function(a, b) { return a.x - b.x; }); var botPts = pts.slice(2, 4).sort(function(a, b) { return b.x - a.x; }); var srcTri = cv.matFromArray(4, 1, cv.CV_32FC2, [ topPts[0].x, topPts[0].y, topPts[1].x, topPts[1].y, botPts[0].x, botPts[0].y, botPts[1].x, botPts[1].y ]); var dstTri = cv.matFromArray(4, 1, cv.CV_32FC2, [0, 0, cfg.warpW, 0, cfg.warpW, cfg.warpH, 0, cfg.warpH]); var M = cv.getPerspectiveTransform(srcTri, dstTri); var warpedColor = new cv.Mat(); cv.warpPerspective(src, warpedColor, M, new cv.Size(cfg.warpW, cfg.warpH)); src.delete(); gray.delete(); thresh.delete(); contours.delete(); hierarchy.delete(); approx.delete(); srcTri.delete(); dstTri.delete(); M.delete(); return { warpedMat: warpedColor, cfg: cfg }; } function matToBase64Jpeg(warpedMat, quality) { quality = quality || 0.85; var canvas = document.createElement('canvas'); canvas.width = warpedMat.cols; canvas.height = warpedMat.rows; cv.imshow(canvas, warpedMat); return canvas.toDataURL('image/jpeg', quality); } function detectOmrId(warpedMat, cfg, fillThreshold) { var gray = new cv.Mat(); cv.cvtColor(warpedMat, gray, cv.COLOR_RGBA2GRAY, 0); var thresh = new cv.Mat(); cv.threshold(gray, thresh, 128, 255, cv.THRESH_BINARY_INV); var detectedId = ''; for (var r = 0; r < 4; r++) { var rowIndex = cfg.idStartRow + r; var markedDigit = '?'; var maxFill = 0; for (var c = 0; c < 10; c++) { var colIndex = cfg.idStartCol + c; var x = (colIndex - OFFSET_COL) * CELL_PX; var y = (rowIndex - OFFSET_ROW) * CELL_PX; var margin = 4; var rect = new cv.Rect(x + margin, y + margin, CELL_PX - margin * 2, CELL_PX - margin * 2); var roi = thresh.roi(rect); var fillRatio = cv.countNonZero(roi) / (roi.cols * roi.rows); roi.delete(); if (fillRatio > fillThreshold && fillRatio > maxFill) { maxFill = fillRatio; markedDigit = String(c); } } detectedId += markedDigit; } gray.delete(); thresh.delete(); return detectedId; } function cropRegionFromBase64(base64, region) { return new Promise(function(resolve) { var img = new Image(); img.onload = function() { var canvas = document.createElement('canvas'); canvas.width = region.width; canvas.height = region.height; var ctx = canvas.getContext('2d'); ctx.drawImage(img, region.x, region.y, region.width, region.height, 0, 0, region.width, region.height); resolve(canvas.toDataURL('image/jpeg', 0.9)); }; img.src = base64.indexOf('data:') === 0 ? base64 : 'data:image/jpeg;base64,' + base64; }); }
